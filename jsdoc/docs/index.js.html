class Person {

    /**
     * 
     * @param {Object} info person에 대한 정보
     */
    constructor(info) {
        /**
         * @property {string} name 이름
         */
        this.name = info.name;
        /**
         * @property { string | number } age 나이 
         */
        this.age = info.age;
    }
    /**
     * @property {Function} greet  A greet with the name
     */
    greet() {
        console.log(`${this.name} says hi!`);
    }
}
